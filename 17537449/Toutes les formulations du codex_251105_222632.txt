# === GENERATE MARKDOWN ===
def generate_md():
    lines = [HEADER]
    global_id = 1
    csv_lines = ["ID,Equation,Category,Tags"]

    for category, equations in FORMS.items():
        lines.append(f"\n---\n\n{category}\n")
        for eq in equations:
            zid = f"Z-{global_id:03d}"
            line = f"{zid} {eq}"
            lines.append(line)
            csv_lines.append(f"{zid},{eq.replace(',', ' ')},{category},zoran,canonical,info-physics")
            global_id += 1

    lines.append(FOOTER)
    content = "\n".join(lines)

    MD_FILE.write_text(content, encoding="utf-8")
    CSV_FILE.write_text("\n".join(csv_lines), encoding="utf-8")
    print(f"✓ Generated: {MD_FILE.name} ({global_id-1} forms)")
    print(f"✓ Generated: {CSV_FILE.name}")

    return content, csv_lines

# === GENERATE SHA512 HASH ===
def generate_hash(md_content, csv_lines):
    csv_content = "\n".join(csv_lines).encode('utf-8')
    combined = md_content.encode('utf-8') + b"\n---\n" + csv_content
